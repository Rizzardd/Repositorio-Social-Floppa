Algoritmo "lista telefonica"

Var

nome: vetor [1..50] de caractere
opcao: caractere
numero: vetor [1..50] de inteiro
contador, limite: inteiro

Inicio
opcao <- "S"
enquanto opcao = "S" faca
contador <- contador +1
escreval ("informe o nome do contato: ")
leia (nome[contador])
escreval ("informe o numero do contato: ")
leia (numero[contador])
escreval ("deseja adicionar mais um contato?")
leia (opcao)
fimenquanto
escreval ("os contatos sÃ£o: ")
limite <- contador
contador <- 1
enquanto (contador <= limite) faca
escreval ("-", nome[contador], ",", numero[contador])
contador <- contador +1
fimenquanto
Fimalgoritmo
